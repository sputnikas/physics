	\item
		\[
			\begin{aligned}
				& g'' = \frac{a}{2g^2} + \frac{b}{g^3} \\
				& \Rightarrow g'^2 = g'^2_0 + \frac{a}{g_0} + \frac{b}{g^2_0} - \frac{a}{g} - \frac{b}{g^2} =
				p - \frac{a}{g} - \frac{b}{g^2} \\
				& \int\limits_{g_0}^{g} \frac{g\,dg}{\sqrt{pg^2 - ag - b}} = t - t_0
			\end{aligned}
